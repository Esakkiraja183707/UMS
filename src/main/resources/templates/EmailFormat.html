import jakarta.mail.internet.MimeMessage; import org.springframework.mail.javamail.MimeMessageHelper; @Service public class EmailService { @Autowired private JavaMailSender mailSender; public void sendHtmlMail(String toEmail) throws Exception { MimeMessage message = mailSender.createMimeMessage(); MimeMessageHelper helper = new MimeMessageHelper(message, true); helper.setTo(toEmail); helper.setSubject("Welcome to Our App 🚀"); String html = """

Welcome 🎉

Your account has been created successfully.

Enjoy using our app 😊

"""; helper.setText(html, true); // true = HTML helper.setFrom("isakkirju@gmail.com"); mailSender.send(message); } }
